```python
def parse_ranges(s):
    result = []
    for part in s.split(', '):
        if '-' in part:
            start, end = map(int, part.split('-'))
            if start <= end:
                result.extend(range(start, end + 1))
            else:
                result.extend(range(start, end - 1, -1))
        else:
            result.append(int(part))
    return result
```
